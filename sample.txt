Arithmetic Program

#include <stdio.h>

// Function prototype
int add(int a, int b);

int main() {
    int num1, num2, result;
    
    // User input
    printf("Enter first number: ");
    scanf("%d", &num1);
    
    printf("Enter second number: ");
    scanf("%d", &num2);
    
    // Perform addition
    result = add(num1, num2);
    printf("Result: %d + %d = %d\n", num1, num2, result);
    
    return 0;
}

// Function definition
int add(int a, int b) {
    return a + b;
}
